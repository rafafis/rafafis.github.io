/************************************************************/
/* Aula 27 - Acelerômetro e Giroscópio                       */
/* Programação do Módulo Acelerômetro e Giroscópio           */
/* MPU6050 que visa controlar dois Servos Motores */
/* através da inclinação do Módulo nas direções dos eixos X e Y. */
/************************************************************/

/* Inclusão das bibliotecas */
#include 
#include 

/* Define os objetos de controle para cada servo */
Servo servo_1;
Servo servo_2;

/* Variáveis para as posições dos servos */
int eixo_X = 0;
int eixo_Y = 0;

/* Define o objeto de controle do módulo em seu endereço I2C */
GY6050 giroscopio(0x68);  // endereço I2C 0x68 padrão para MPU6050

void setup() {
  /* Inicializa o módulo */
  giroscopio.initialisation();
  delay(100);

  /* Define as portas digitais de controle para cada servo */
  servo_1.attach(7);
  servo_2.attach(8);
}

void loop() {
  /* Coleta os dados do módulo e converte em ângulos entre 0 e 180 graus */
  /* Atualiza as variáveis com os ângulos atuais */
  eixo_X = map( giroscopio.refresh('A', 'X'), -90, 90, 0, 180 );
  eixo_Y = map( giroscopio.refresh('A', 'Y'), -90, 90, 0, 180 );

  /* Posiciona cada servo no ângulo atual */
  servo_1.write(eixo_X);
  servo_2.write(eixo_Y);

  /* Pequena pausa para atualizar as posições */
  delay(15);
}
