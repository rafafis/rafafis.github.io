/* Programa: Utilizando Joystick Shield */
#include <Servo.h>

/* Objetos dos servos */
Servo servo1;
Servo servo2;

/* Pinos dos servos */
const int Servo1 = A2;
const int Servo2 = A3;

/* Pinos dos botÃµes */
const int Botao_A = 2;
const int Botao_B = 3;
const int Botao_C = 4;
const int Botao_D = 5;
const int Botao_E = 6;
const int Botao_F = 7;
const int Botao_K = 8;

/* Pinos dos LEDs */
const int LED_C = 9;
const int LED_D = 10;
const int LED_E = 11;
const int LED_F = 12;
const int Buzzer = 13;

const int LED_A = A4;
const int LED_B = A5;

/* Pinos dos potenciÃ´metros */
const int Pot_X = A0;
const int Pot_Y = A1;

void setup() {
  servo1.attach(Servo1);
  servo2.attach(Servo2);

  for (int i = 2; i <= 8; i++) {
    pinMode(i, INPUT_PULLUP);
  }

  for (int i = 9; i <= 13; i++) {
    pinMode(i, OUTPUT);
  }

  pinMode(LED_A, OUTPUT);
  pinMode(LED_B, OUTPUT);
}

void loop() {
  // Controle dos servos com os potenciÃ´metros
  servo1.write(map(analogRead(Pot_X), 0, 1023, 0, 180));
  servo2.write(map(analogRead(Pot_Y), 0, 1023, 0, 180));

  // LEDs e Buzzer com base nos botÃµes
  digitalWrite(LED_A, !digitalRead(Botao_A));
  digitalWrite(LED_B, !digitalRead(Botao_B));
  digitalWrite(LED_C, !digitalRead(Botao_C));
  digitalWrite(LED_D, !digitalRead(Botao_D));
  digitalWrite(LED_E, !digitalRead(Botao_E));
  digitalWrite(LED_F, !digitalRead(Botao_F));

  if (!digitalRead(Botao_K)) {
    tone(Buzzer, 800);
  } else {
    noTone(Buzzer);
  }
}
