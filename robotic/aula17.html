// Inclui a biblioteca de controle do teclado

// Inclui a biblioteca do servo motor

// Cria um objeto servo para controle do servo
Servo servo1;

// Senha para destrancar a fechadura

// Pinos que est√£o conectados os LEDs

  {'1', '2', '3', 'A'},

  {'*', '0', '#', 'D'}
};

void loop() {
  // Captura a tecla pressionada
  char key = tec.getKey();

  // Se nenhuma tecla foi pressionada, sai da fun√ß√£o
  if (!key) return;

  // Se a tecla pressionada for '*' ou '#', reinicia a tentativa

    return;

  // Se a tecla pressionada estiver correta em rela√ß√£o √† senha

// Fun√ß√£o que mant√©m a fechadura trancada

// Fun√ß√£o que mant√©m a fechadura destrancada
