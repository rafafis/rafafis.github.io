/************************************************************/
/* Aula 27 - Sensor LDR                                     */
/* Programação: Aciona um LED a partir da leitura do Sensor */
/* de luminosidade (LDR).                                   */
/************************************************************/

int Sensor_LDR;        // para armazenar leitura analógica
const int Pin_LDR = A0;  // pino analógico (A0) onde o LDR está conectado
const int Pin_LED = 5;   // pino digital para controle do LED

// ajuste este valor conforme sua iluminação / experimento
const int limiar = 800;  // se a leitura for menor que esse valor, aciona LED

void setup() {
  Serial.begin(9600);
  pinMode(Pin_LED, OUTPUT);
  // Não é necessário definir pinMode para o pino analógico ao usar analogRead
}

void loop() {
  // lê valor analógico (0 a 1023) correspondente à luminosidade
  Sensor_LDR = analogRead(Pin_LDR);

  Serial.print("Valor lido: ");
  Serial.println(Sensor_LDR);

  delay(100);  // espera pequeno intervalo antes da próxima leitura

  // se a luminosidade for baixa (valor abaixo do limiar), aciona LED
  if (Sensor_LDR > limiar) {
    digitalWrite(Pin_LED, HIGH);
  } else {
    digitalWrite(Pin_LED, LOW);
  }
}
