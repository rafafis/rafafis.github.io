Aula 16 - Display 7 segmentos com 4 dígitos

Ligação:

Display TM1637 Arduino Função
VCC 5V Alimentação
GND GND Terra
DIO Pino 3 Dados (I/O)
CLK Pino 2 Clock

Código da Aula:


#include <TM1637Display.h>

// Define os pinos CLK e DIO
#define CLK 2
#define DIO 3

// Cria o objeto display
TM1637Display display(CLK, DIO);

void setup() {
  display.setBrightness(0x0f); // Brilho de 0x00 (mínimo) até 0x0f (máximo)
}

void loop() {
  for (int i = 0; i <= 9999; i++) {
    display.showNumberDec(i, true); // true para preencher com zeros à esquerda
    delay(500);
  }
}
